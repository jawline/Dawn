00053 /* F, G and H are basic MD5 functions: selection, majority, parity */
00054 #define F(x, y, z) (((x) & (y)) | ((~x) & (z)))
00055 #define G(x, y, z) (((x) & (z)) | ((y) & (~z)))
00056 #define H(x, y, z) ((x) ^ (y) ^ (z))
00057 #define I(x, y, z) ((y) ^ ((x) | (~z)))

00059 /* ROTATE_LEFT rotates x left n bits */
